\subsection{Кубический сенсор}
Система наблюдения:
\begin{equation}\nonumber
\left\lbrace\begin{array}{l}
x_t = \left(\begin{array}{c}
\frac{x_0}{1 + x_0^2} \\
\frac{x_1}{1 + x_1^2} \\
\end{array}\right)
 + \left(\begin{array}{cc}
1 & 0 \\
0 & 1 \\
\end{array}\right)
 w_t, \quad t= 0,\ldots,T, \quad x_0 = \eta, \\
y_t = \left(\begin{array}{c}
x_0^3+x_0 \\
x_1^3+x_1 \\
\end{array}\right)
 + \nu_t.
\end{array}\right.
\end{equation}

$w_t \sim \mathcal{N}\left(\mathbf{0}, \mathbf{E}\right)$, 
$\nu_t \sim \mathcal{N}\left(\mathbf{0}, \mathbf{E}\right)$, 
$\eta_t \sim \mathcal{N}\left(\left(\begin{array}{c}
100 \\
100 \\
\end{array}\right)
, \left(\begin{array}{cc}
100 & 0 \\
0 & 100 \\
\end{array}\right)
\right)$
